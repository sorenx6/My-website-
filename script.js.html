const audio = document.getElementById("audio"); const playBtn = document.getElementById("playBtn"); let isPlaying = false; playBtn.addEventListener("click", () => { if (isPlaying) { audio.pause(); playBtn.innerHTML = "▶ Play"; } else { audio.play(); playBtn.innerHTML = "⏸ Pause"; } isPlaying = !isPlaying; }); // Audio khatam hone par button reset audio.addEventListener("ended", () => { playBtn.innerHTML = "▶ Play"; isPlaying = false; });
